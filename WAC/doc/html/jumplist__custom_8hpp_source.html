112 if (file.good())
113 {
114 file.unsetf(std::ios::skipws);
115 file.seekg(0, std::ios::end);
116 const size_t size = file.tellg();

122 //récupération des dates
123 HANDLE hFile = CreateFile(_path.wstring().c_str(), // name of the write
124 GENERIC_READ, // open for reading
125 0, // do not share
126 NULL, // default security
127 OPEN_EXISTING, // open existing file only
128 FILE_ATTRIBUTE_NORMAL, // normal file
129 NULL); // no attr. template
130 if (hFile != INVALID_HANDLE_VALUE) {
131 FILE_BASIC_INFO fileInfo;
132 GetFileInformationByHandleEx(hFile, FileBasicInfo, &fileInfo, sizeof(FILE_BASIC_INFO));
133 memcpy(&createdUtc, &fileInfo.CreationTime, sizeof(createdUtc));
134 memcpy(&modifiedUtc, &fileInfo.LastWriteTime, sizeof(modifiedUtc));
135 memcpy(&accessedUtc, &fileInfo.LastAccessTime, sizeof(accessedUtc));

143 std::wstring baseName = _path.filename().wstring().substr(0, p); // nom de fichier sans extension
144 application = ansi_to_utf8(from_appId(baseName));
145
146 typeInt = bytes_to_unsigned_int(buffer);
147 switch (typeInt) {
148 case 0: {
149 _errors->push_back({ replaceAll(path,L"\\",L"\\\\") + L" : Custom category not supported",ERROR_UNIDENTIFIED_ERROR});

166 if ((size > 24) && (typeInt == 2)) {
