359 oleParser(LPBYTE _buffer, size_t _bufferSize) {
360 buffer = _buffer;
361 bufferSize = _bufferSize;
362
363 // 0. Process header
364 header = oleHeader(buffer, _bufferSize);
365
366 //Big Files
367 if (header.MSATFirstSectorId > -2) {
368 int maxSlotsPerBlock = header.sectorSize / 4;
369 int remainingSlots = header.totalSATSectors - 109; // 109 for header part already done
370 int remainingByteLen = 4 * remainingSlots;
371 int msatOffset = (header.MSATFirstSectorId + 1) * header.sectorSize;
372 int startOffset = 0;
373 LPBYTE remainingBytes = new BYTE[remainingByteLen];
374 while (remainingSlots > 0) {
375 if (remainingSlots > maxSlotsPerBlock) {
376 // in this case we have to only take so many
377 for (int x = 0; x < header.sectorSize - 4; x++) {
378 remainingBytes[x] = buffer[msatOffset + x];
379 }
380 remainingSlots -= maxSlotsPerBlock - 1;
381 int newOffset = *reinterpret_cast<int*>(buffer + msatOffset + (4 * (maxSlotsPerBlock - 1)));
382 msatOffset = (newOffset + 1) * header.sectorSize;
383 startOffset += (maxSlotsPerBlock - 1) * 4;
384 }
385 else {
386 //copy it and be done with it
387 for (int x = 0; x < remainingSlots * 4; x++) {
388 remainingBytes[startOffset + x] = buffer[msatOffset + x];
389 remainingSlots -= remainingSlots;
390 }
391 }
392 }
393
394 remainingSlots = header.totalSATSectors - 109;
395
396 for (int i = 0; i < remainingSlots; i += 4)
397 {
398 int sectorId = *reinterpret_cast<int*>(remainingBytes + i * 4) * header.sectorSize + 512; // 512 is for the header
399 header.SATSectors.push_back(sectorId);
400 }
401 delete[] remainingBytes;
402 }
403
404 //We need to get all the bytes that make up the SectorAllocationTable
405 //start with empty array to hold our bytes
406
407
408 for (int sector : header.SATSectors)
409 {
410 if (sector + header.sectorSize > _bufferSize)
411 throw std::length_error("file corrupt - Error copying data from the Sector Allocation Table");
412
413 //fill the Sat
414 for (int x = 0; x < header.sectorSize; x += 4) { // a chaque "sector" on copie sectorSize entiers
415 log(3, L"ðŸ”ˆ*reinterpret_cast<int*> sat");
416 sat.push_back(*reinterpret_cast<int*>(buffer + sector + x));
417 }
418 }
419
420 //Just as with the SAT, but this time, with the SmallSectorAllocationTable
421 if (header.SSATFirstSectorId != -2)
422 {
423 log(3, L"ðŸ”ˆGetIntFromSat ssat");
424 ssat = GetIntFromSat(header.SSATFirstSectorId);
425 }
426
427 // 1. Process all Directory entries
428 // https://github.com/EricZimmerman/OleCf/blob/master/OleCf/OleCfFile.cs#L138
429
430 log(3, L"ðŸ”ˆGetBytesFromSat dirBytes");
431 std::vector<BYTE> dirBytes = GetBytesFromSat(header.directoryStreamFirstSectorId);
432 LPBYTE pDirBytes = &dirBytes[0];
433 int dirIndex = 0;
434 if (dirIndex + 128 > dirBytes.size())
435 throw std::length_error("file corrupt - Error copying data from directory index");
436
437 while (dirIndex < dirBytes.size())
438 {
439 log(3, L"ðŸ”ˆ*reinterpret_cast<short int*> dirLen");
440 int dirLen = *reinterpret_cast<short int*>(pDirBytes + dirIndex + 64);
441 if (pDirBytes[dirIndex + 66] != 0 && dirLen > 0) { //0 is empty directory structure
442 log(3, L"ðŸ”ˆDirectory d");
443 Directory d = Directory(pDirBytes + dirIndex);
444 directories.push_back(d);
445 }
446 dirIndex += 128;
447 }

449 //the Root Entry directory item contains all the sectors we need for small sector stuff, so get the data and cut it up so we can use it later
450 //when we are done we will have a list of byte arrays, each 64 bytes long, that we can string together later based on SSAT
451
