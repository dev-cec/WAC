253 oleHeader(LPBYTE buffer, size_t _bufferSize) {
254 signature = bytes_to_unsigned_long(buffer);
255 if (signature != _signature)
256 throw std::runtime_error("bad signature");
257
258 littleIndian = (bytes_to_short(buffer + 28) == (short)0xfffe); //0xfeff = big indian
259 if (littleIndian == false) {
260 throw std::runtime_error("Big indian Format not Handle, please handle this file specifically");
261 }
262
263 sectorSize = pow(2, bytes_to_short(buffer + 30)); // Sector size at offset 30, en puissance de 2
264 shortSectorSize = pow(2, bytes_to_short(buffer + 32)); // Short sector size at offset 32, en puissance de 2
265 totalSATSectors = bytes_to_int(buffer + 44); // Total Sector Allocation Table(SAT) sectors at offset 44
266 directoryStreamFirstSectorId = bytes_to_int(buffer + 48); // Sector ID of first sector used by Directory at offset 48
267 minimumStandardStreamSize = bytes_to_unsigned_int(buffer + 56); // Minimum size of a standard stream in bytes at offset 56
268 SSATFirstSectorId = bytes_to_int(buffer + 60); // Sector ID of the first sector used for the Short Sector Allocation Table(SSAT) at offset 60
269 totalSSATSectors = bytes_to_unsigned_int(buffer + 64); // Total sectors used for SSAT at offset 64
270 MSATFirstSectorId = bytes_to_int(buffer + 68);
271 MSATTotalSectors = bytes_to_int(buffer + 72);
272 // Process MSAT
273 if (_bufferSize < 516)
274 throw std::length_error("File corrupt - file smaller than header size"); // header = 76 + 109*4 + 4
275
276 for (int i = 0; i < 109; i++) {
277 int addr = bytes_to_int(buffer + 76 + i * 4);
278
279 if (addr >= 0) {
280 if (i < totalSATSectors)
281 SATSectors.push_back(addr * sectorSize + 512); // 512 is for the header
282 else
283 throw std::length_error("File corrupt - The total number of sectors was larger than the one expected from the header data");
284 }
285 }

333 oleParser(LPBYTE _buffer, int _bufferSize) {
334 buffer = _buffer;
335 bufferSize = _bufferSize;
336
337 // 0. Process header
338 header = oleHeader(buffer, _bufferSize);
339
340
341 //Big Files
342 if (header.MSATFirstSectorId > -2) {
343 int maxSlotsPerBlock = header.sectorSize / 4;
344 int remainingSlots = header.totalSATSectors - 109; // 109 for header part already done
345 int remainingByteLen = 4 * remainingSlots;
346 int msatOffset = (header.MSATFirstSectorId + 1) * header.sectorSize;
347 int startOffset = 0;
348 LPBYTE remainingBytes = new BYTE[remainingByteLen];
349 while (remainingSlots > 0) {
350 if (remainingSlots > maxSlotsPerBlock) {
351 // in this case we have to only take so many
352 for (int x = 0; x < header.sectorSize - 4; x++) {
353 remainingBytes[x] = buffer[msatOffset + x];
354 }
355 remainingSlots -= maxSlotsPerBlock - 1;
356 int newOffset = bytes_to_int(buffer + msatOffset + (4 * (maxSlotsPerBlock - 1)));
357 msatOffset = (newOffset + 1) * header.sectorSize;
358 startOffset += (maxSlotsPerBlock - 1) * 4;
359 }
360 else {
361 //copy it and be done with it
362 for (int x = 0; x < remainingSlots * 4; x++) {
363 remainingBytes[startOffset + x] = buffer[msatOffset + x];
364 remainingSlots -= remainingSlots;
365 }
366 }
367 }
368
369 remainingSlots = header.totalSATSectors - 109;
370
371 for (int i = 0; i < remainingSlots; i += 4)
372 {
373 int sectorId = bytes_to_int(remainingBytes + i * 4) * header.sectorSize + 512; // 512 is for the header
374 header.SATSectors.push_back(sectorId);
375 }
376 }
377
378 //We need to get all the bytes that make up the SectorAllocationTable
379 //start with empty array to hold our bytes
380
381
382 for (int sector : header.SATSectors)
383 {
384 if (sector + header.sectorSize > _bufferSize)
385 throw std::length_error("file corrupt - Error copying data from the Sector Allocation Table");
386
387
388 //fill the Sat
389 for (int x = 0; x < header.sectorSize; x += 4) { // a chaque "sector" on copie sectorSize entiers
390 sat.push_back(bytes_to_int(buffer + sector + x));
391 }
392 }
393
394 //Just as with the SAT, but this time, with the SmallSectorAllocationTable
395 if (header.SSATFirstSectorId != -2)
396 {
397 ssat = GetIntFromSat(header.SSATFirstSectorId);
398 }
399
400 // 1. Process all Directory entries
401 // https://github.com/EricZimmerman/OleCf/blob/master/OleCf/OleCfFile.cs#L138
402
403 std::vector<BYTE> dirBytes = GetBytesFromSat(header.directoryStreamFirstSectorId);
404 LPBYTE pDirBytes = &dirBytes[0];
405 int dirIndex = 0;
406 if (dirIndex + 128 > dirBytes.size())
407 throw std::length_error("file corrupt - Error copying data from directory index");
408
409 while (dirIndex < dirBytes.size())
410 {
411 int dirLen = bytes_to_short(pDirBytes + dirIndex + 64);
412 if (pDirBytes[dirIndex + 66] != 0 && dirLen > 0) { //0 is empty directory structure
413 Directory d = Directory(pDirBytes + dirIndex);
414 directories.push_back(d);
415 }
416 dirIndex += 128;
417 }
418
419 //the Root Entry directory item contains all the sectors we need for small sector stuff, so get the data and cut it up so we can use it later
420 //when we are done we will have a list of byte arrays, each 64 bytes long, that we can string together later based on SSAT
421
422 rootEntry = findDirectory(L"root entry");
423 if (rootEntry.name != L"" && rootEntry.directorySize > 0) {
424 std::vector<BYTE> b = GetBytesFromSat((int)rootEntry.firstSectorID);
425 int shortIndex = 0;
426 while (shortIndex < b.size())
427 {
428 std::vector<BYTE> shortChunk;
429 if (shortIndex + header.shortSectorSize > b.size())
430 throw std::length_error("file corrupt - Error copying data for short sector");
431
432 for (int x = 0; x < header.shortSectorSize; x++)
433 shortChunk.push_back(b[shortIndex + x]);
434 shortSectors.push_back(shortChunk);
435 shortIndex += header.shortSectorSize;
436 }
437 }
438 };
