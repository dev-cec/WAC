245 RecentDoc(std::filesystem::path _path, std::wstring _sid, bool _debug, bool _dump, std::vector<std::tuple<std::wstring, HRESULT>>* errors)
246 {
247 //Parsing
248 Sid = _sid;
249 //path retourne un codage ANSI mais on veut de l'UTF8
250 path = ansi_to_utf8(_path.wstring());
251 path = replaceAll(path, L"\\", L"\\\\");//escape \ in std::string
252 target = L"";
253 if (_path.extension() == ".lnk" || _path.extension() == ".LNK") {
254 std::ifstream file(_path.wstring(), std::ios::binary);
255 if (file.good())
256 {
257 file.unsetf(std::ios::skipws);
258 file.seekg(0, std::ios::end);
259 const size_t size = file.tellg();
260 file.seekg(0, std::ios::beg);
261 LPBYTE buffer = new BYTE[size];
262 file.read(reinterpret_cast<CHAR*>(buffer), size);
263 file.close();
264
265 parseLNK(buffer, _debug, _dump, errors);
266
267 }
268 }
269 if (_path.extension() == ".url" || _path.extension() == ".URL") {
270 std::ifstream file(_path);
271 std::string line;
272 if (file.is_open()) {
273 getline(file, line); //skip first line
274 getline(file, line);
275 line = ansi_to_utf8(line); //ensure utf8
276 line = line.substr(4);//suppression de URL= en début de ligne
277 line = decodeURIComponent(line);
278 target = string_to_wstring(line);
279 file.close();
280 }
281 }
282
283 //récupération des dates
284 HANDLE hFile = CreateFile(_path.wstring().c_str(), // name of the write
285 GENERIC_READ, // open for reading
286 0, // do not share
287 NULL, // default security
288 OPEN_EXISTING, // open existing file only
289 FILE_ATTRIBUTE_NORMAL, // normal file
290 NULL); // no attr. template
291 if (hFile != INVALID_HANDLE_VALUE) {
292 FILE_BASIC_INFO fileInfo;
293 GetFileInformationByHandleEx(hFile, FileBasicInfo, &fileInfo, sizeof(FILE_BASIC_INFO));
294 memcpy(&sourceCreatedUtc, &fileInfo.CreationTime, sizeof(sourceCreatedUtc));
295 memcpy(&sourceModifiedUtc, &fileInfo.LastWriteTime, sizeof(sourceModifiedUtc));
296 memcpy(&sourceAccessedUtc, &fileInfo.LastAccessTime, sizeof(sourceAccessedUtc));
297 FileTimeToLocalFileTime(&sourceCreatedUtc, &sourceCreated);
298 FileTimeToLocalFileTime(&sourceModifiedUtc, &sourceModified);
299 FileTimeToLocalFileTime(&sourceAccessedUtc, &sourceAccessed);
300 }
